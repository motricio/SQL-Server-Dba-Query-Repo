SELECT 
    LOGINAME AS LOGINNAME,
    PROGRAM_NAME,
    DB_NAME(DBID) AS DBNAME,
    COUNT(DBID) AS CONNECTIONS,
    --HOSTNAME,
    --LOGIN_TIME,
    MIN(LOGIN_TIME)MIN_LOGIN_TIME,
    STATUS
FROM
    SYS.SYSPROCESSES
WHERE 
    DBID > 6
GROUP BY
	DBID,LOGINAME,STATUS,PROGRAM_NAME
ORDER BY LOGINNAME
